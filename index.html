Credit Card Mask


Usually when you buy something, you're asked whether your credit card number, phone number or answer to your most secret question is still correct. However, since someone could look over your shoulder, you don't want that shown on your screen. Instead, we mask it.

function maskify(cc) {

                    const lenStr = cc.length;
                    let result ='';
                    
                    if (lenStr > 4) {
                      let lattice = lenStr - 4;
                      for (let i = 0; i < lattice; i = i + 1){
                        result = result + '#';
                      }
                      result = result + cc[lenStr - 4] + cc[lenStr - 3] + cc[lenStr - 2] + cc[lenStr - 1];
                    } else {
                      switch (lenStr) {
                          case 4: result = result + cc[lenStr - 4] + cc[lenStr - 3] + cc[lenStr - 2] + cc[lenStr - 1];
                          break;
                          case 3: result = result + cc[lenStr - 3] + cc[lenStr - 2] + cc[lenStr - 1];
                          break;
                          case 2: result = result + cc[lenStr - 2] + cc[lenStr - 1];
                          break;
                          case 1: result = result + cc[lenStr - 1];
                          break;
                      }
                    }
                  return result;
                  
                  }
